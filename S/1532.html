  51 const char *zlog_get_level_name(int log_level) /* {{{ */
  52 {
  53         if (log_level < 0) {

  55         } else if (log_level < ZLOG_DEBUG || log_level > ZLOG_ALERT) {
  56                 return "unknown value";
  57         }

 101 void zlog_ex(const char *function, int line, int flags, const char *fmt, ...) /* {{{ */
 102 {
 103         struct timeval tv;
 104         char buf[MAX_LINE_LENGTH];
 105         const size_t buf_size = MAX_LINE_LENGTH;
 106         va_list args;
 107         size_t len = 0;
 108         int truncated = 0;
 109         int saved_errno;
 110 
 111         if (external_logger) {
 112                 va_start(args, fmt);
 113                 len = vsnprintf(buf, buf_size, fmt, args);
 114                 va_end(args);
 115                 if (len >= buf_size) {
 116                         memcpy(buf + buf_size - sizeof("..."), "...", sizeof("...") - 1);
 117                         len = buf_size - 1;
 118                 }
 119                 external_logger(flags & ZLOG_LEVEL_MASK, buf, len);
 120                 len = 0;
 121                 memset(buf, '\0', buf_size);
 122         }
 123 
 124         if ((flags & ZLOG_LEVEL_MASK) < zlog_level) {
 125                 return;
 126         }
 127 
 128         saved_errno = errno;
 129 #ifdef HAVE_SYSLOG_H
 130         if (zlog_fd == ZLOG_SYSLOG /* && !fpm_globals.is_child */) {
 131                 len = 0;
 132                 if (zlog_level == ZLOG_DEBUG) {
 133                         len += snprintf(buf, buf_size, "[%s] %s(), line %d: ", level_names[flags & ZLOG_LEVEL_MASK], function, line);
 134                 } else {
 135                         len += snprintf(buf, buf_size, "[%s] ", level_names[flags & ZLOG_LEVEL_MASK]);
 136                 }
 137         } else
 138 #endif
 139         {
 140                 if (!fpm_globals.is_child) {
 141                         gettimeofday(&tv, 0);
 142                         len = zlog_print_time(&tv, buf, buf_size);
 143                 }
 144                 if (zlog_level == ZLOG_DEBUG) {
 145                         if (!fpm_globals.is_child) {
 146                                 len += snprintf(buf + len, buf_size - len, "%s: pid %d, %s(), line %d: ", level_names[flags & ZLOG_LEVEL_MASK], getpid(), function, line);
 147                         } else {
 148                                 len += snprintf(buf + len, buf_size - len, "%s: %s(), line %d: ", level_names[flags & ZLOG_LEVEL_MASK], function, line);
 149                         }
 150                 } else {
 151                         len += snprintf(buf + len, buf_size - len, "%s: ", level_names[flags & ZLOG_LEVEL_MASK]);
 152                 }
 153         }
 154 
 155         if (len > buf_size - 1) {
 156                 truncated = 1;
 157         }
 158 
 159         if (!truncated) {
 160                 va_start(args, fmt);
 161                 len += vsnprintf(buf + len, buf_size - len, fmt, args);
 162                 va_end(args);
 163                 if (len >= buf_size) {
 164                         truncated = 1;
 165                 }
 166         }
 167 
 168         if (!truncated) {
 169                 if (flags & ZLOG_HAVE_ERRNO) {
 170                         len += snprintf(buf + len, buf_size - len, ": %s (%d)", strerror(saved_errno), saved_errno);
 171                         if (len >= buf_size) {
 172                                 truncated = 1;
 173                         }
 174                 }
 175         }
 176 
 177         if (truncated) {
 178                 memcpy(buf + buf_size - sizeof("..."), "...", sizeof("...") - 1);
 179                 len = buf_size - 1;
 180         }
 181 
 182 #ifdef HAVE_SYSLOG_H
 183         if (zlog_fd == ZLOG_SYSLOG) {
 184                 buf[len] = '\0';
 185                 php_syslog(syslog_priorities[zlog_level], "%s", buf);
 186                 buf[len++] = '\n';
 187         } else 
 188 #endif
 189         {
 190                 buf[len++] = '\n';
 191                 write(zlog_fd > -1 ? zlog_fd : STDERR_FILENO, buf, len);
 192         }
 193 
 194         if (zlog_fd != STDERR_FILENO && zlog_fd != -1 && !launched && (flags & ZLOG_LEVEL_MASK) >= ZLOG_NOTICE) {
 195                 write(STDERR_FILENO, buf, len);
 196         }
 197 }
