 158 static PHP_INI_MH(UpdateDefaultFilter) /* {{{ */
 159 {
 160         int i, size = sizeof(filter_list) / sizeof(filter_list_entry);
 161 
 162         for (i = 0; i < size; ++i) {
 163                 if ((strcasecmp(new_value, filter_list[i].name) == 0)) {
 164                         IF_G(default_filter) = filter_list[i].id;
 165                         return SUCCESS;
 166                 }
 167         }
 168         /* Fallback to the default filter */
 169         IF_G(default_filter) = FILTER_DEFAULT;
 170         return SUCCESS;
 171 }

 330 static filter_list_entry php_find_filter(long id) /* {{{ */
 331 {
 332         int i, size = sizeof(filter_list) / sizeof(filter_list_entry);
 333 
 334         for (i = 0; i < size; ++i) {
 335                 if (filter_list[i].id == id) {
 336                         return filter_list[i];
 337                 }
 338         }
 339         /* Fallback to "string" filter */
 340         for (i = 0; i < size; ++i) {
 341                 if (filter_list[i].id == FILTER_DEFAULT) {
 342                         return filter_list[i];
 343                 }
 344         }
 345         /* To shut up GCC */
 346         return filter_list[0];
 347 }

 361 static void php_zval_filter(zval **value, long filter, long flags, zval *options, char* charset, zend_bool copy TSRMLS_DC) /* {{{ */
 362 {
 363         filter_list_entry  filter_func;
 364 
 365         filter_func = php_find_filter(filter);
 366 
 367         if (!filter_func.id) {
 368                 /* Find default filter */
 369                 filter_func = php_find_filter(FILTER_DEFAULT);
 370         }
 371 
 372         if (copy) {
 373                 SEPARATE_ZVAL(value);
 374         }
 375 
 376         /* #49274, fatal error with object without a toString method
 377           Fails nicely instead of getting a recovarable fatal error. */
 378         if (Z_TYPE_PP(value) == IS_OBJECT) {
 379                 zend_class_entry *ce;
 380 
 381                 ce = Z_OBJCE_PP(value);
 382                 if (!ce->__tostring) {
 383                         ZVAL_FALSE(*value);
 384                         return;
 385                 }
 386         }
 387 
 388         /* Here be strings */
 389         convert_to_string(*value);
 390 
 391         filter_func.function(*value, flags, options, charset TSRMLS_CC);
 392 
 393         if (
 394                 options && (Z_TYPE_P(options) == IS_ARRAY || Z_TYPE_P(options) == IS_OBJECT) &&
 395                 ((flags & FILTER_NULL_ON_FAILURE && Z_TYPE_PP(value) == IS_NULL) || 
 396                 (!(flags & FILTER_NULL_ON_FAILURE) && Z_TYPE_PP(value) == IS_BOOL && Z_LVAL_PP(value) == 0)) &&
 397                 zend_hash_exists(HASH_OF(options), "default", sizeof("default"))
 398         ) {
 399                 zval **tmp; 
 400                 if (zend_hash_find(HASH_OF(options), "default", sizeof("default"), (void **)&tmp) == SUCCESS) {
 401                         MAKE_COPY_ZVAL(tmp, *value);
 402                 }
 403         }
 404 }

 407 static unsigned int php_sapi_filter(int arg, char *var, char **val, unsigned int val_len, unsigned int *new_val_len TSRMLS_DC) /* {{{ */
 408 {
 409         zval  new_var, raw_var;
 410         zval *array_ptr = NULL, *orig_array_ptr = NULL;
 411         int retval = 0;
 412 
 413         assert(*val != NULL);
 414 
 415 #define PARSE_CASE(s,a,t)                    \
 416                 case s:                              \
 417                         if (!IF_G(a)) {                  \
 418                                 ALLOC_ZVAL(array_ptr);       \
 419                                 array_init(array_ptr);       \
 420                                 INIT_PZVAL(array_ptr);       \
 421                                 IF_G(a) = array_ptr;         \
 422                         } else {                         \
 423                                 array_ptr = IF_G(a);         \
 424                         }                                \
 425                         orig_array_ptr = PG(http_globals)[t]; \
 426                         break;
 427 
 428         switch (arg) {
 429                 PARSE_CASE(PARSE_POST,    post_array,    TRACK_VARS_POST)
 430                 PARSE_CASE(PARSE_GET,     get_array,     TRACK_VARS_GET)
 431                 PARSE_CASE(PARSE_COOKIE,  cookie_array,  TRACK_VARS_COOKIE)
 432                 PARSE_CASE(PARSE_SERVER,  server_array,  TRACK_VARS_SERVER)
 433                 PARSE_CASE(PARSE_ENV,     env_array,     TRACK_VARS_ENV)
 434 
 435                 case PARSE_STRING: /* PARSE_STRING is used by parse_str() function */
 436                         retval = 1;
 437                         break;
 438         }
 439 
 440         /* 
 441          * According to rfc2965, more specific paths are listed above the less specific ones.
 442          * If we encounter a duplicate cookie name, we should skip it, since it is not possible
 443          * to have the same (plain text) cookie name for the same path and we should not overwrite
 444          * more specific cookies with the less specific ones.
 445         */
 446         if (arg == PARSE_COOKIE && orig_array_ptr && zend_symtable_exists(Z_ARRVAL_P(orig_array_ptr), var, strlen(var)+1)) {
 447                 return 0;
 448         }
 449 
 450         if (array_ptr) {
 451                 /* Store the RAW variable internally */
 452                 Z_STRLEN(raw_var) = val_len;
 453                 Z_STRVAL(raw_var) = estrndup(*val, val_len);
 454                 Z_TYPE(raw_var) = IS_STRING;
 455 
 456                 php_register_variable_ex(var, &raw_var, array_ptr TSRMLS_CC);
 457         }
 458 
 459         if (val_len) {
 460                 /* Register mangled variable */
 461                 Z_STRLEN(new_var) = val_len;
 462                 Z_TYPE(new_var) = IS_STRING;
 463 
 464                 if (IF_G(default_filter) != FILTER_UNSAFE_RAW) {
 465                         zval *tmp_new_var = &new_var;
 466                         Z_STRVAL(new_var) = estrndup(*val, val_len);
 467                         INIT_PZVAL(tmp_new_var);
 468                         php_zval_filter(&tmp_new_var, IF_G(default_filter), IF_G(default_filter_flags), NULL, NULL/*charset*/, 0 TSRMLS_CC);
 469                 } else {
 470                         Z_STRVAL(new_var) = estrndup(*val, val_len);
 471                 }
 472         } else { /* empty string */
 473                 ZVAL_EMPTY_STRING(&new_var);
 474         }
 475 
 476         if (orig_array_ptr) {
 477                 php_register_variable_ex(var, &new_var, orig_array_ptr TSRMLS_CC);
 478         }
 479 
 480         if (retval) {
 481                 if (new_val_len) {
 482                         *new_val_len = Z_STRLEN(new_var);
 483                 }
 484                 efree(*val);
 485                 if (Z_STRLEN(new_var)) {
 486                         *val = estrndup(Z_STRVAL(new_var), Z_STRLEN(new_var));
 487                 } else {
 488                         *val = estrdup("");
 489                 }
 490                 zval_dtor(&new_var);
 491         }
 492 
 493         return retval;
 494 }

 497 static void php_zval_filter_recursive(zval **value, long filter, long flags, zval *options, char *charset, zend_bool copy TSRMLS_DC) /* {{{ */
 498 {
 499         if (Z_TYPE_PP(value) == IS_ARRAY) {
 500                 zval **element;
 501                 HashPosition pos;
 502 
 503                 if (Z_ARRVAL_PP(value)->nApplyCount > 1) {
 504                         return;
 505                 }
 506 
 507                 for (zend_hash_internal_pointer_reset_ex(Z_ARRVAL_PP(value), &pos);
 508                          zend_hash_get_current_data_ex(Z_ARRVAL_PP(value), (void **) &element, &pos) == SUCCESS;
 509                          zend_hash_move_forward_ex(Z_ARRVAL_PP(value), &pos)
 510                 ) {
 511                         SEPARATE_ZVAL_IF_NOT_REF(element);
 512                         if (Z_TYPE_PP(element) == IS_ARRAY) {
 513                                 Z_ARRVAL_PP(element)->nApplyCount++;
 514                                 php_zval_filter_recursive(element, filter, flags, options, charset, copy TSRMLS_CC);
 515                                 Z_ARRVAL_PP(element)->nApplyCount--;
 516                         } else {
 517                                 php_zval_filter(element, filter, flags, options, charset, copy TSRMLS_CC);
 518                         }
 519                 }
 520         } else {
 521                 php_zval_filter(value, filter, flags, options, charset, copy TSRMLS_CC);
 522         }
 523 }

 591 static void php_filter_call(zval **filtered, long filter, zval **filter_args, const int copy, long filter_flags TSRMLS_DC) /* {{{ */
 592 {
 593         zval  *options = NULL;
 594         zval **option;
 595         char  *charset = NULL;
 596 
 597         if (filter_args && Z_TYPE_PP(filter_args) != IS_ARRAY) {
 598                 long lval;
 599 
 600                 PHP_FILTER_GET_LONG_OPT(filter_args, lval);
 601 
 602                 if (filter != -1) { /* handler for array apply */
 603                         /* filter_args is the filter_flags */
 604                         filter_flags = lval;
 605 
 606                         if (!(filter_flags & FILTER_REQUIRE_ARRAY ||  filter_flags & FILTER_FORCE_ARRAY)) {
 607                                 filter_flags |= FILTER_REQUIRE_SCALAR;
 608                         }
 609                 } else {
 610                         filter = lval;
 611                 }
 612         } else if (filter_args) {
 613                 if (zend_hash_find(HASH_OF(*filter_args), "filter", sizeof("filter"), (void **)&option) == SUCCESS) {
 614                         PHP_FILTER_GET_LONG_OPT(option, filter);
 615                 }
 616 
 617                 if (zend_hash_find(HASH_OF(*filter_args), "flags", sizeof("flags"), (void **)&option) == SUCCESS) {
 618                         PHP_FILTER_GET_LONG_OPT(option, filter_flags);
 619 
 620                         if (!(filter_flags & FILTER_REQUIRE_ARRAY ||  filter_flags & FILTER_FORCE_ARRAY)) {
 621                                 filter_flags |= FILTER_REQUIRE_SCALAR;
 622                         }
 623                 }
 624 
 625                 if (zend_hash_find(HASH_OF(*filter_args), "options", sizeof("options"), (void **)&option) == SUCCESS) {
 626                         if (filter != FILTER_CALLBACK) {
 627                                 if (Z_TYPE_PP(option) == IS_ARRAY) {
 628                                         options = *option;
 629                                 }
 630                         } else {
 631                                 options = *option;
 632                                 filter_flags = 0;
 633                         }
 634                 }
 635         }
 636 
 637         if (Z_TYPE_PP(filtered) == IS_ARRAY) {
 638                 if (filter_flags & FILTER_REQUIRE_SCALAR) {
 639                         if (copy) {
 640                                 SEPARATE_ZVAL(filtered);
 641                         }
 642                         zval_dtor(*filtered);
 643                         if (filter_flags & FILTER_NULL_ON_FAILURE) {
 644                                 ZVAL_NULL(*filtered);
 645                         } else {
 646                                 ZVAL_FALSE(*filtered);
 647                         }
 648                         return;
 649                 }
 650                 php_zval_filter_recursive(filtered, filter, filter_flags, options, charset, copy TSRMLS_CC);
 651                 return;
 652         }
 653         if (filter_flags & FILTER_REQUIRE_ARRAY) {
 654                 if (copy) {
 655                         SEPARATE_ZVAL(filtered);
 656                 }
 657                 zval_dtor(*filtered);
 658                 if (filter_flags & FILTER_NULL_ON_FAILURE) {
 659                         ZVAL_NULL(*filtered);
 660                 } else {
 661                         ZVAL_FALSE(*filtered);
 662                 }
 663                 return;
 664         }
 665 
 666         php_zval_filter(filtered, filter, filter_flags, options, charset, copy TSRMLS_CC);
 667         if (filter_flags & FILTER_FORCE_ARRAY) {
 668                 zval *tmp;
 669 
 670                 ALLOC_ZVAL(tmp);
 671                 MAKE_COPY_ZVAL(filtered, tmp);
 672 
 673                 zval_dtor(*filtered);
 674 
 675                 array_init(*filtered);
 676                 add_next_index_zval(*filtered, tmp);
 677         }
 678 }

 681 static void php_filter_array_handler(zval *input, zval **op, zval *return_value, zend_bool add_empty TSRMLS_DC) /* {{{ */
 682 {
 683         char *arg_key;
 684         uint arg_key_len;
 685         ulong index;
 686         HashPosition pos;
 687         zval **tmp, **arg_elm;
 688 
 689         if (!op) {
 690                 zval_dtor(return_value);
 691                 MAKE_COPY_ZVAL(&input, return_value);
 692                 php_filter_call(&return_value, FILTER_DEFAULT, NULL, 0, FILTER_REQUIRE_ARRAY TSRMLS_CC);
 693         } else if (Z_TYPE_PP(op) == IS_LONG) {
 694                 zval_dtor(return_value);
 695                 MAKE_COPY_ZVAL(&input, return_value);
 696                 php_filter_call(&return_value, Z_LVAL_PP(op), NULL, 0, FILTER_REQUIRE_ARRAY TSRMLS_CC);
 697         } else if (Z_TYPE_PP(op) == IS_ARRAY) {
 698                 array_init(return_value);
 699 
 700                 zend_hash_internal_pointer_reset(Z_ARRVAL_PP(op));
 701                 for (zend_hash_internal_pointer_reset_ex(Z_ARRVAL_PP(op), &pos);
 702                         zend_hash_get_current_data_ex(Z_ARRVAL_PP(op), (void **) &arg_elm, &pos) == SUCCESS;
 703                         zend_hash_move_forward_ex(Z_ARRVAL_PP(op), &pos))
 704                 {
 705                         if (zend_hash_get_current_key_ex(Z_ARRVAL_PP(op), &arg_key, &arg_key_len, &index, 0, &pos) != HASH_KEY_IS_STRING) {
 706                                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "Numeric keys are not allowed in the definition array");
 707                                 zval_dtor(return_value);
 708                                 RETURN_FALSE;
 709                         }
 710                         if (arg_key_len < 2) {
 711                                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "Empty keys are not allowed in the definition array");
 712                                 zval_dtor(return_value);
 713                                 RETURN_FALSE;
 714                         }
 715                         if (zend_hash_find(Z_ARRVAL_P(input), arg_key, arg_key_len, (void **)&tmp) != SUCCESS) {
 716                                 if (add_empty) {
 717                                         add_assoc_null_ex(return_value, arg_key, arg_key_len);
 718                                 }
 719                         } else {
 720                                 zval *nval;
 721 
 722                                 ALLOC_ZVAL(nval);
 723                                 MAKE_COPY_ZVAL(tmp, nval);
 724 
 725                                 php_filter_call(&nval, -1, arg_elm, 0, FILTER_REQUIRE_SCALAR TSRMLS_CC);
 726                                 add_assoc_zval_ex(return_value, arg_key, arg_key_len, nval);
 727                         }
 728                 }
 729         } else {
 730                 RETURN_FALSE;
 731         }
 732 }

 735 /* {{{ proto mixed filter_input(constant type, string variable_name [, long filter [, mixed options]])
 736  * Returns the filtered variable 'name'* from source `type`.
 737  */
 738 PHP_FUNCTION(filter_input)
 739 {
 740         long   fetch_from, filter = FILTER_DEFAULT;
 741         zval **filter_args = NULL, **tmp;
 742         zval  *input = NULL;
 743         char *var;
 744         int var_len;
 745 
 746         if (zend_parse_parameters(ZEND_NUM_ARGS() TSRMLS_CC, "ls|lZ", &fetch_from, &var, &var_len, &filter, &filter_args) == FAILURE) {
 747                 return;
 748         }
 749 
 750         if (!PHP_FILTER_ID_EXISTS(filter)) {
 751                 RETURN_FALSE;
 752         }
 753 
 754         input = php_filter_get_storage(fetch_from TSRMLS_CC);
 755 
 756         if (!input || !HASH_OF(input) || zend_hash_find(HASH_OF(input), var, var_len + 1, (void **)&tmp) != SUCCESS) {
 757                 long filter_flags = 0;
 758                 zval **option, **opt, **def;
 759                 if (filter_args) {
 760                         if (Z_TYPE_PP(filter_args) == IS_LONG) {
 761                                 filter_flags = Z_LVAL_PP(filter_args);
 762                         } else if (Z_TYPE_PP(filter_args) == IS_ARRAY && zend_hash_find(HASH_OF(*filter_args), "flags", sizeof("flags"), (void **)&option) == SUCCESS) {
 763                                 PHP_FILTER_GET_LONG_OPT(option, filter_flags);
 764                         }
 765                         if (Z_TYPE_PP(filter_args) == IS_ARRAY && 
 766                                 zend_hash_find(HASH_OF(*filter_args), "options", sizeof("options"), (void **)&opt) == SUCCESS &&
 767                                 Z_TYPE_PP(opt) == IS_ARRAY &&
 768                                 zend_hash_find(HASH_OF(*opt), "default", sizeof("default"), (void **)&def) == SUCCESS
 769                         ) {
 770                                 MAKE_COPY_ZVAL(def, return_value);
 771                                 return;
 772                         }
 773                 }
 774 
 775                 /* The FILTER_NULL_ON_FAILURE flag inverts the usual return values of
 776                  * the function: normally when validation fails false is returned, and
 777                  * when the input value doesn't exist NULL is returned. With the flag
 778                  * set, NULL and false should be returned, respectively. Ergo, although
 779                  * the code below looks incorrect, it's actually right. */
 780                 if (filter_flags & FILTER_NULL_ON_FAILURE) {
 781                         RETURN_FALSE;
 782                 } else {
 783                         RETURN_NULL();
 784                 }
 785         }
 786 
 787         MAKE_COPY_ZVAL(tmp, return_value);
 788 
 789         php_filter_call(&return_value, filter, filter_args, 1, FILTER_REQUIRE_SCALAR TSRMLS_CC);
 790 }
