 101 #define MYSQLI_RETURN_LONG_LONG(__val) \
 102 { \
 103         if ((__val) < LONG_MAX) {               \
 104                 RETURN_LONG((long) (__val));            \
 105         } else {                                \
 106                 char *ret;                      \
 107                 /* always used with my_ulonglong -> %llu */ \
 108                 int l = spprintf(&ret, 0, MYSQLI_LLU_SPEC, (__val));    \
 109                 RETURN_STRINGL(ret, l, 0);              \
 110         }                                       \
 111 }

 141 #define MYSQLI_REPORT_MYSQL_ERROR(mysql) \
 142 if ((MyG(report_mode) & MYSQLI_REPORT_ERROR) && mysql_errno(mysql)) { \
 143         php_mysqli_report_error(mysql_sqlstate(mysql), mysql_errno(mysql), mysql_error(mysql) TSRMLS_CC); \
 144 }
 145 
 146 #define MYSQLI_REPORT_STMT_ERROR(stmt) \
 147 if ((MyG(report_mode) & MYSQLI_REPORT_ERROR) && mysql_stmt_errno(stmt)) { \
 148         php_mysqli_report_error(mysql_stmt_sqlstate(stmt), mysql_stmt_errno(stmt), mysql_stmt_error(stmt) TSRMLS_CC); \
 149 }
