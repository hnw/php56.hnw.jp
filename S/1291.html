  40 ZIP_EXTERN int
  41 zip_file_replace(struct zip *za, zip_uint64_t idx, struct zip_source *source, zip_flags_t flags)
  42 {
  43     if (idx >= za->nentry || source == NULL) {
  44         _zip_error_set(&za->error, ZIP_ER_INVAL, 0);
  45         return -1;
  46     }
  47 
  48     if (_zip_file_replace(za, idx, NULL, source, flags) == -1)
  49         return -1;
  50 
  51     return 0;
  52 }

  58 
  59 zip_int64_t
  60 _zip_file_replace(struct zip *za, zip_uint64_t idx, const char *name, struct zip_source *source, zip_flags_t flags)
  61 {
  62     zip_uint64_t za_nentry_prev;
  63     
  64     if (ZIP_IS_RDONLY(za)) {
  65         _zip_error_set(&za->error, ZIP_ER_RDONLY, 0);
  66         return -1;
  67     }
  68 
  69     za_nentry_prev = za->nentry;
  70     if (idx == ZIP_UINT64_MAX) {
  71         zip_int64_t i = -1;
  72         
  73         if (flags & ZIP_FL_OVERWRITE)
  74             i = _zip_name_locate(za, name, flags, NULL);
  75 
  76         if (i == -1) {
  77             /* create and use new entry, used by zip_add */
  78             if ((i=_zip_add_entry(za)) < 0)
  79                 return -1;
  80         }
  81         idx = (zip_uint64_t)i;
  82     }
  83     
  84     if (name && _zip_set_name(za, idx, name, flags) != 0) {
  85         if (za->nentry != za_nentry_prev) {
  86             _zip_entry_finalize(za->entry+idx);
  87             za->nentry = za_nentry_prev;
  88         }
  89         return -1;
  90     }
  91 
  92     /* does not change any name related data, so we can do it here;
  93      * needed for a double add of the same file name */
  94     _zip_unchange_data(za->entry+idx);
  95 
  96     if (za->entry[idx].orig != NULL && (za->entry[idx].changes == NULL || (za->entry[idx].changes->changed & ZIP_DIRENT_COMP_METHOD) == 0)) {
  97         if (za->entry[idx].changes == NULL) {
  98             if ((za->entry[idx].changes=_zip_dirent_clone(za->entry[idx].orig)) == NULL) {
  99                 _zip_error_set(&za->error, ZIP_ER_MEMORY, 0);
 100                 return -1;
 101             }
 102         }
 103 
 104         za->entry[idx].changes->comp_method = ZIP_CM_REPLACED_DEFAULT;
 105         za->entry[idx].changes->changed |= ZIP_DIRENT_COMP_METHOD;
 106     }
 107         
 108     za->entry[idx].source = source;
 109 
 110     return (zip_int64_t)idx;
 111 }
