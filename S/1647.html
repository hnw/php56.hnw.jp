 688                                                         }

 692         {                                                                                                                                                               \
 693                 char *_name = (name);                                                                                                           \
 694                                                                                                                                                                         \
 695                 ZEND_SET_SYMBOL_WITH_LENGTH(symtable, _name, strlen(_name)+1, var, 1, 0);       \

 698 #define ZEND_SET_SYMBOL_WITH_LENGTH(symtable, name, name_length, var, _refcount, _is_ref)                               \
 699         {                                                                                                                                                                                                       \
 700                 zval **orig_var;                                                                                                                                                                \
 701                                                                                                                                                                                                                 \
 702                 if (zend_hash_find(symtable, (name), (name_length), (void **) &orig_var)==SUCCESS                               \
 703                         && PZVAL_IS_REF(*orig_var)) {                                                                                                                           \
 704                         Z_SET_REFCOUNT_P(var, Z_REFCOUNT_PP(orig_var));                                                                                         \
 705                         Z_SET_ISREF_P(var);                                                                                                                                                     \
 706                                                                                                                                                                                                                 \
 707                         if (_refcount) {                                                                                                                                                        \
 708                                 Z_SET_REFCOUNT_P(var, Z_REFCOUNT_P(var) + _refcount - 1);                                                               \
 709                         }                                                                                                                                                                                       \
 710                         zval_dtor(*orig_var);                                                                                                                                           \
 711                         **orig_var = *(var);                                                                                                                                            \
 712                         FREE_ZVAL(var);                                                                                                                                                         \
 713                 } else {                                                                                                                                                                                \
 714                         Z_SET_ISREF_TO_P(var, _is_ref);                                                                                                                         \
 715                         if (_refcount) {                                                                                                                                                        \
 716                                 Z_SET_REFCOUNT_P(var, _refcount);                                                                                                               \
 717                         }                                                                                                                                                                                       \
 718                         zend_hash_update(symtable, (name), (name_length), &(var), sizeof(zval *), NULL);                        \
 719                 }                                                                                                                                                                                               \
 720         }
