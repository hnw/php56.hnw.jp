  97         /* create initial children in all pools */

  99                 int is_parent;

 104                         goto run_child;
 105                 }

 107                 /* handle error */

 110                         fpm_event_loop(1);
 111                 }
 112         }
 113 
 114         /* run event loop forever */
 115         fpm_event_loop(0);
