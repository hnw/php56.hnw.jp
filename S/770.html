 527                 return;
 528         }
 529         MYSQLI_FETCH_RESOURCE_CONN(mysql, &mysql_link, MYSQLI_STATUS_VALID);
 530 
 531         MYSQLI_ENABLE_MQ;
 532         if (mysql_real_query(mysql->mysql, query, query_len)) {
 533 #ifndef MYSQLI_USE_MYSQLND
 534                 char s_error[MYSQL_ERRMSG_SIZE], s_sqlstate[SQLSTATE_LENGTH+1];

 540                 s_errno = mysql_errno(mysql->mysql);
 541 #else

 543 #endif
 544                 MYSQLI_REPORT_MYSQL_ERROR(mysql->mysql);
 545                 MYSQLI_DISABLE_MQ;
 546 
 547 #ifndef MYSQLI_USE_MYSQLND
 548                 /* restore error information */
 549                 strcpy(mysql->mysql->net.last_error, s_error);
 550                 strcpy(mysql->mysql->net.sqlstate, s_sqlstate);
 551                 mysql->mysql->net.last_errno = s_errno;

 561 /* {{{ proto mixed mysqli_query(object link, string query [,int resultmode]) */
 562 PHP_FUNCTION(mysqli_query)
 563 {
 564         MY_MYSQL                        *mysql;
 565         zval                            *mysql_link;
 566         MYSQLI_RESOURCE         *mysqli_resource;
 567         MYSQL_RES                       *result = NULL;
 568         char                            *query = NULL;
 569         int                             query_len;
 570         long                            resultmode = MYSQLI_STORE_RESULT;
 571 
 572         if (zend_parse_method_parameters(ZEND_NUM_ARGS() TSRMLS_CC, getThis(), "Os|l", &mysql_link, mysqli_link_class_entry, &query, &query_len, &resultmode) == FAILURE) {
 573                 return;
 574         }
 575 
 576         if (!query_len) {
 577                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "Empty query");
 578                 RETURN_FALSE;
 579         }
 580 #ifdef MYSQLI_USE_MYSQLND
 581         if ((resultmode & ~MYSQLI_ASYNC) != MYSQLI_USE_RESULT && (resultmode & ~(MYSQLI_ASYNC | MYSQLI_STORE_RESULT_COPY_DATA)) != MYSQLI_STORE_RESULT) {
 582 #else
 583         if ((resultmode & ~MYSQLI_ASYNC) != MYSQLI_USE_RESULT && (resultmode & ~MYSQLI_ASYNC) != MYSQLI_STORE_RESULT) {
 584 #endif
 585                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "Invalid value for resultmode");
 586                 RETURN_FALSE;
 587         }
 588 
 589         MYSQLI_FETCH_RESOURCE_CONN(mysql, &mysql_link, MYSQLI_STATUS_VALID);
 590 
 591         MYSQLI_DISABLE_MQ;
 592 
 593 
 594 #ifdef MYSQLI_USE_MYSQLND
 595         if (resultmode & MYSQLI_ASYNC) {
 596                 if (mysqli_async_query(mysql->mysql, query, query_len)) {
 597                         MYSQLI_REPORT_MYSQL_ERROR(mysql->mysql);
 598                         RETURN_FALSE;
 599                 }
 600                 mysql->async_result_fetch_type = resultmode & ~MYSQLI_ASYNC;
 601                 RETURN_TRUE;
 602         }
 603 #endif
 604 
 605         if (mysql_real_query(mysql->mysql, query, query_len)) {
 606                 MYSQLI_REPORT_MYSQL_ERROR(mysql->mysql);
 607                 RETURN_FALSE;
 608         }
 609 
 610         if (!mysql_field_count(mysql->mysql)) {
 611                 /* no result set - not a SELECT */
 612                 if (MyG(report_mode) & MYSQLI_REPORT_INDEX) {
 613                         php_mysqli_report_index(query, mysqli_server_status(mysql->mysql) TSRMLS_CC);
 614                 }
 615                 RETURN_TRUE;
 616         }
 617 
 618 #ifdef MYSQLI_USE_MYSQLND
 619         switch (resultmode & ~(MYSQLI_ASYNC | MYSQLI_STORE_RESULT_COPY_DATA)) {
 620 #else
 621         switch (resultmode & ~MYSQLI_ASYNC) {
 622 #endif
 623                 case MYSQLI_STORE_RESULT:
 624 #ifdef MYSQLI_USE_MYSQLND
 625                         if (resultmode & MYSQLI_STORE_RESULT_COPY_DATA) {
 626                                 result = mysqlnd_store_result_ofs(mysql->mysql);
 627                         } else
 628 #endif
 629                                 result = mysql_store_result(mysql->mysql);
 630                         break;
 631                 case MYSQLI_USE_RESULT:
 632                         result = mysql_use_result(mysql->mysql);
 633                         break;
 634         }
 635         if (!result) {
 636                 php_mysqli_throw_sql_exception((char *)mysql_sqlstate(mysql->mysql), mysql_errno(mysql->mysql) TSRMLS_CC,
 637                                                                                 "%s", mysql_error(mysql->mysql));
 638                 RETURN_FALSE;
 639         }
 640 
 641         if (MyG(report_mode) & MYSQLI_REPORT_INDEX) {
 642                 php_mysqli_report_index(query, mysqli_server_status(mysql->mysql) TSRMLS_CC);
 643         }
 644 
 645         mysqli_resource = (MYSQLI_RESOURCE *)ecalloc (1, sizeof(MYSQLI_RESOURCE));
 646         mysqli_resource->ptr = (void *)result;
 647         mysqli_resource->status = MYSQLI_STATUS_VALID;
 648         MYSQLI_RETURN_RESOURCE(mysqli_resource, mysqli_result_class_entry);
 649 }

 785 /* {{{ proto int mysqli_poll(array read, array write, array error, long sec [, long usec]) U
 786    Poll connections */
 787 PHP_FUNCTION(mysqli_poll)
 788 {
 789         zval                    *r_array, *e_array, *dont_poll_array;
 790         MYSQLND                 **new_r_array = NULL, **new_e_array = NULL, **new_dont_poll_array = NULL;
 791         long                    sec = 0, usec = 0;
 792         enum_func_status ret;
 793         int                     desc_num;
 794 
 795         if (zend_parse_parameters(ZEND_NUM_ARGS() TSRMLS_CC, "a!a!al|l", &r_array, &e_array, &dont_poll_array, &sec, &usec) == FAILURE) {
 796                 return;
 797         }
 798         if (sec < 0 || usec < 0) {
 799                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "Negative values passed for sec and/or usec");
 800                 RETURN_FALSE;
 801         }
 802 
 803         if (!r_array && !e_array) {
 804                 php_error_docref(NULL TSRMLS_CC, E_WARNING, "No stream arrays were passed");
 805                 RETURN_FALSE;
 806         }
 807 
 808         if (r_array != NULL) {
 809                 mysqlnd_zval_array_to_mysqlnd_array(r_array, &new_r_array TSRMLS_CC);
 810         }
 811         if (e_array != NULL) {
 812                 mysqlnd_zval_array_to_mysqlnd_array(e_array, &new_e_array TSRMLS_CC);
 813         }
 814 
 815         ret = mysqlnd_poll(new_r_array, new_e_array, &new_dont_poll_array, sec, usec, &desc_num);
 816 
 817         mysqlnd_dont_poll_zval_array_from_mysqlnd_array(r_array != NULL ? new_dont_poll_array:NULL, r_array, dont_poll_array TSRMLS_CC);
 818 
 819         if (r_array != NULL) {
 820                 mysqlnd_zval_array_from_mysqlnd_array(new_r_array, r_array TSRMLS_CC);
 821         }
 822         if (e_array != NULL) {
 823                 mysqlnd_zval_array_from_mysqlnd_array(new_e_array, e_array TSRMLS_CC);
 824         }
 825 
 826         if (new_dont_poll_array) {
 827                 efree(new_dont_poll_array);
 828         }
 829         if (new_r_array) {
 830                 efree(new_r_array);
 831         }
 832         if (new_e_array) {
 833                 efree(new_e_array);
 834         }
 835         if (ret == PASS) {
 836                 RETURN_LONG(desc_num);
 837         } else {
 838                 RETURN_FALSE;
 839         }
 840 }
