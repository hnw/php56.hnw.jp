  29 static inline zend_uint phpdbg_decode_literal(zend_op_array *ops, zend_literal *literal TSRMLS_DC) /* {{{ */
  30 {
  31         int iter = 0;
  32 
  33         while (iter < ops->last_literal) {
  34                 if (literal == &ops->literals[iter]) {
  35                         return iter;
  36                 }
  37                 iter++;
  38         }
  39 
  40         return 0;
  41 } /* }}} */

  43 static inline char *phpdbg_decode_op(zend_op_array *ops, znode_op *op, zend_uint type, HashTable *vars TSRMLS_DC) /* {{{ */
  44 {
  45         char *decode = NULL;
  46 
  47         switch (type &~ EXT_TYPE_UNUSED) {
  48                 case IS_CV:
  49                         asprintf(&decode, "$%s", ops->vars[op->var].name);
  50                 break;
  51 
  52                 case IS_VAR:
  53                 case IS_TMP_VAR: {
  54                         zend_ulong id = 0, *pid = NULL;
  55                         if (vars != NULL) {
  56                                 if (zend_hash_index_find(vars, (zend_ulong) ops->vars - op->var, (void**) &pid) != SUCCESS) {
  57                                         id = zend_hash_num_elements(vars);
  58                                         zend_hash_index_update(
  59                                                 vars, (zend_ulong) ops->vars - op->var,
  60                                                 (void**) &id,
  61                                                 sizeof(zend_ulong), NULL);
  62                                 } else id = *pid;
  63                         }
  64                         asprintf(&decode, "@%lu", id);
  65                 } break;
  66 
  67                 case IS_CONST:
  68                         asprintf(&decode, "C%u", phpdbg_decode_literal(ops, op->literal TSRMLS_CC));
  69                 break;
  70 
  71                 case IS_UNUSED:
  72                         asprintf(&decode, "<unused>");
  73                 break;
  74         }
  75         return decode;
  76 } /* }}} */

  78 char *phpdbg_decode_opline(zend_op_array *ops, zend_op *op, HashTable *vars TSRMLS_DC) /*{{{ */
  79 {
  80         char *decode[4] = {NULL, NULL, NULL, NULL};
  81 
  82         switch (op->opcode) {
  83         case ZEND_JMP:
  84 #ifdef ZEND_GOTO
  85         case ZEND_GOTO:
  86 #endif
  87 #ifdef ZEND_FAST_CALL
  88         case ZEND_FAST_CALL:
  89 #endif
  90                         asprintf(&decode[1], "J%ld", op->op1.jmp_addr - ops->opcodes);
  91                 goto format;
  92 
  93         case ZEND_JMPZNZ:
  94                         decode[1] = phpdbg_decode_op(ops, &op->op1, op->op1_type, vars TSRMLS_CC);
  95                         asprintf(
  96                                 &decode[2], "J%u or J%lu", op->op2.opline_num, op->extended_value);
  97                 goto result;
  98 
  99         case ZEND_JMPZ:
 100         case ZEND_JMPNZ:
 101         case ZEND_JMPZ_EX:
 102         case ZEND_JMPNZ_EX:
 103 
 104 #ifdef ZEND_JMP_SET
 105         case ZEND_JMP_SET:
 106 #endif
 107 #ifdef ZEND_JMP_SET_VAR
 108         case ZEND_JMP_SET_VAR:
 109 #endif
 110                 decode[1] = phpdbg_decode_op(ops, &op->op1, op->op1_type, vars TSRMLS_CC);
 111                 asprintf(
 112                         &decode[2], "J%ld", op->op2.jmp_addr - ops->opcodes);
 113         goto result;
 114 
 115         case ZEND_RECV_INIT:
 116                 goto result;
 117 
 118                 default: {
 119                         decode[1] = phpdbg_decode_op(ops, &op->op1, op->op1_type, vars TSRMLS_CC);
 120                         decode[2] = phpdbg_decode_op(ops, &op->op2, op->op2_type, vars TSRMLS_CC);
 121 result:
 122                         decode[3] = phpdbg_decode_op(ops, &op->result, op->result_type, vars TSRMLS_CC);
 123 format:
 124                         asprintf(
 125                                 &decode[0],
 126                                 "%-20s %-20s %-20s",
 127                                 decode[1] ? decode[1] : "",
 128                                 decode[2] ? decode[2] : "",
 129                                 decode[3] ? decode[3] : "");
 130                 }
 131         }
 132 
 133         if (decode[1])
 134                 free(decode[1]);
 135         if (decode[2])
 136                 free(decode[2]);
 137         if (decode[3])
 138                 free(decode[3]);
 139 
 140         return decode[0];
 141 } /* }}} */

 143 void phpdbg_print_opline_ex(zend_execute_data *execute_data, HashTable *vars, zend_bool ignore_flags TSRMLS_DC) /* {{{ */
 144 {
 145         /* force out a line while stepping so the user knows what is happening */
 146         if (ignore_flags ||
 147                 (!(PHPDBG_G(flags) & PHPDBG_IS_QUIET) ||
 148                 (PHPDBG_G(flags) & PHPDBG_IS_STEPPING) ||
 149                 (PHPDBG_G(oplog)))) {
 150 
 151                 zend_op *opline = execute_data->opline;
 152                 char *decode = phpdbg_decode_opline(execute_data->op_array, opline, vars TSRMLS_CC);
 153 
 154                 if (ignore_flags || (!(PHPDBG_G(flags) & PHPDBG_IS_QUIET) || (PHPDBG_G(flags) & PHPDBG_IS_STEPPING))) {
 155                         /* output line info */
 156                         phpdbg_notice("L%-5u %16p %-30s %s %s",
 157                            opline->lineno,
 158                            opline,
 159                            phpdbg_decode_opcode(opline->opcode),
 160                            decode,
 161                            execute_data->op_array->filename ? execute_data->op_array->filename : "unknown");
 162                 }
 163 
 164                 if (!ignore_flags && PHPDBG_G(oplog)) {
 165                         phpdbg_log_ex(PHPDBG_G(oplog), "L%-5u %16p %-30s %s %s",
 166                                 opline->lineno,
 167                                 opline,
 168                                 phpdbg_decode_opcode(opline->opcode),
 169                                 decode,
 170                                 execute_data->op_array->filename ? execute_data->op_array->filename : "unknown");
 171                 }
 172 
 173                 if (decode) {
 174                         free(decode);
 175                 }
 176         }
 177 } /* }}} */
