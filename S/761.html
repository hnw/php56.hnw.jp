 118 /* {{{ php_free_stmt_bind_buffer */
 119 void php_free_stmt_bind_buffer(BIND_BUFFER bbuf, int type)
 120 {
 121         unsigned int i;
 122 
 123         if (!bbuf.var_cnt) {
 124                 return;
 125         }
 126 
 127         for (i=0; i < bbuf.var_cnt; i++) {
 128 
 129                 /* free temporary bind buffer */
 130                 if (type == FETCH_RESULT && bbuf.buf[i].val) {
 131                         efree(bbuf.buf[i].val);
 132                 }
 133 
 134                 if (bbuf.vars[i]) {
 135                         zval_ptr_dtor(&bbuf.vars[i]);
 136                 }
 137         }
 138 
 139         if (bbuf.vars) {
 140                 efree(bbuf.vars);
 141         }
 142 
 143         /*
 144           Don't free bbuf.is_null for FETCH_RESULT since we have allocated
 145           is_null and buf in one block so we free only buf, which is the beginning
 146           of the block. When FETCH_SIMPLE then buf wasn't allocated together with
 147           buf and we have to free it.
 148         */
 149         if (type == FETCH_RESULT) {
 150                 efree(bbuf.buf);
 151         } else if (type == FETCH_SIMPLE){
 152                 efree(bbuf.is_null);
 153         }
 154 
 155         bbuf.var_cnt = 0;
 156 }

 321 zval *mysqli_read_property(zval *object, zval *member, int type, const zend_literal *key TSRMLS_DC)
 322 {
 323         zval tmp_member;
 324         zval *retval;
 325         mysqli_object *obj;
 326         mysqli_prop_handler *hnd;
 327         int ret;
 328 
 329         ret = FAILURE;
 330         obj = (mysqli_object *)zend_objects_get_address(object TSRMLS_CC);
 331 
 332         if (member->type != IS_STRING) {
 333                 tmp_member = *member;
 334                 zval_copy_ctor(&tmp_member);
 335                 convert_to_string(&tmp_member);
 336                 member = &tmp_member;
 337         }
 338 
 339         if (obj->prop_handler != NULL) {
 340                 ret = zend_hash_find(obj->prop_handler, Z_STRVAL_P(member), Z_STRLEN_P(member)+1, (void **) &hnd);
 341         }
 342 
 343         if (ret == SUCCESS) {
 344                 ret = hnd->read_func(obj, &retval TSRMLS_CC);
 345                 if (ret == SUCCESS) {
 346                         /* ensure we're creating a temporary variable */
 347                         Z_SET_REFCOUNT_P(retval, 0);
 348                 } else {
 349                         retval = EG(uninitialized_zval_ptr);
 350                 }
 351         } else {
 352                 zend_object_handlers * std_hnd = zend_get_std_object_handlers();
 353                 retval = std_hnd->read_property(object, member, type, key TSRMLS_CC);
 354         }
 355 
 356         if (member == &tmp_member) {
 357                 zval_dtor(member);
 358         }
 359         return(retval);

 486 /* {{{ mysqli_objects_new
 487  */
 488 PHP_MYSQLI_EXPORT(zend_object_value) mysqli_objects_new(zend_class_entry *class_type TSRMLS_DC)
 489 {
 490         zend_object_value retval;
 491         mysqli_object *intern;
 492         zend_class_entry *mysqli_base_class;
 493         zend_objects_free_object_storage_t free_storage;
 494 
 495         intern = emalloc(sizeof(mysqli_object));
 496         memset(intern, 0, sizeof(mysqli_object));
 497         intern->ptr = NULL;
 498         intern->prop_handler = NULL;
 499 
 500         mysqli_base_class = class_type;
 501         while (mysqli_base_class->type != ZEND_INTERNAL_CLASS &&
 502                    mysqli_base_class->parent != NULL) {
 503                 mysqli_base_class = mysqli_base_class->parent;
 504         }
 505         zend_hash_find(&classes, mysqli_base_class->name, mysqli_base_class->name_length + 1,
 506                                         (void **) &intern->prop_handler);
 507 
 508         zend_object_std_init(&intern->zo, class_type TSRMLS_CC);
 509         object_properties_init(&intern->zo, class_type);
 510 
 511         /* link object */
 512         if (instanceof_function(class_type, mysqli_link_class_entry TSRMLS_CC)) {
 513                 free_storage = mysqli_link_free_storage;
 514         } else if (instanceof_function(class_type, mysqli_driver_class_entry TSRMLS_CC)) { /* driver object */
 515                 free_storage = mysqli_driver_free_storage;
 516         } else if (instanceof_function(class_type, mysqli_stmt_class_entry TSRMLS_CC)) { /* stmt object */
 517                 free_storage = mysqli_stmt_free_storage;
 518         } else if (instanceof_function(class_type, mysqli_result_class_entry TSRMLS_CC)) { /* result object */
 519                 free_storage = mysqli_result_free_storage;
 520         } else if (instanceof_function(class_type, mysqli_warning_class_entry TSRMLS_CC)) { /* warning object */
 521                 free_storage = mysqli_warning_free_storage;
 522         } else {
 523                 free_storage = mysqli_objects_free_storage;
 524         }
 525 
 526         retval.handle = zend_objects_store_put(intern, (zend_objects_store_dtor_t) zend_objects_destroy_object, free_storage, NULL TSRMLS_CC);
 527         retval.handlers = &mysqli_object_handlers;
 528 
 529         return retval;
 530 }
