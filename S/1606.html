  55 PHPAPI int flock(int fd, int op)
  56 {
  57         HANDLE hdl = (HANDLE) _get_osfhandle(fd);
  58         DWORD low = 1, high = 0;
  59         OVERLAPPED offset =
  60         {0, 0, 0, 0, NULL};
  61         if (hdl < 0)
  62                 return -1;                              /* error in file descriptor */
  63         /* bug for bug compatible with Unix */
  64         UnlockFileEx(hdl, 0, low, high, &offset);
  65         switch (op & ~LOCK_NB) {        /* translate to LockFileEx() op */
  66                 case LOCK_EX:                   /* exclusive */
  67                         if (LockFileEx(hdl, LOCKFILE_EXCLUSIVE_LOCK +
  68                                                 ((op & LOCK_NB) ? LOCKFILE_FAIL_IMMEDIATELY : 0),
  69                                                    0, low, high, &offset))
  70                                 return 0;
  71                         break;
  72                 case LOCK_SH:                   /* shared */
  73                         if (LockFileEx(hdl, ((op & LOCK_NB) ? LOCKFILE_FAIL_IMMEDIATELY : 0),
  74                                                    0, low, high, &offset))
  75                                 return 0;
  76                         break;
  77                 case LOCK_UN:                   /* unlock */
  78                         return 0;                       /* always succeeds */
  79                 default:                                /* default */
  80                         break;
  81         }
  82         errno = EINVAL;                         /* bad call */
  83         return -1;
  84 }
