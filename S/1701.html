1301 /* Copy to YYRES the contents of YYSTR after stripping away unnecessary
1302    quotes and backslashes, so that it's suitable for yyerror.  The
1303    heuristic is that double-quoting is unnecessary unless the string
1304    contains an apostrophe, a comma, or backslash (other than
1305    backslash-backslash).  YYSTR is taken from yytname.  If YYRES is
1306    null, do not copy; instead, return the length of what the result
1307    would have been.  */
1308 static YYSIZE_T zend_yytnamerr(char *yyres, const char *yystr)
1309 {
1310         if (!yyres) {
1311                 return yystrlen(yystr);
1312         }
1313         {
1314                 TSRMLS_FETCH();
1315                 if (CG(parse_error) == 0) {
1316                         char buffer[120];
1317                         const unsigned char *end, *str, *tok1 = NULL, *tok2 = NULL;
1318                         unsigned int len = 0, toklen = 0, yystr_len;
1319                         
1320                         CG(parse_error) = 1;
1321 
1322                         if (LANG_SCNG(yy_text)[0] == 0 &&
1323                                 LANG_SCNG(yy_leng) == 1 &&
1324                                 memcmp(yystr, "\"end of file\"", sizeof("\"end of file\"") - 1) == 0) {
1325                                 yystpcpy(yyres, "end of file");
1326                                 return sizeof("end of file")-1;
1327                         }
1328                         
1329                         str = LANG_SCNG(yy_text);
1330                         end = memchr(str, '\n', LANG_SCNG(yy_leng));
1331                         yystr_len = yystrlen(yystr);
1332                         
1333                         if ((tok1 = memchr(yystr, '(', yystr_len)) != NULL
1334                                 && (tok2 = zend_memrchr(yystr, ')', yystr_len)) != NULL) {
1335                                 toklen = (tok2 - tok1) + 1;
1336                         } else {
1337                                 tok1 = tok2 = NULL;
1338                                 toklen = 0;
1339                         }
1340                         
1341                         if (end == NULL) {
1342                                 len = LANG_SCNG(yy_leng) > 30 ? 30 : LANG_SCNG(yy_leng);
1343                         } else {
1344                                 len = (end - str) > 30 ? 30 : (end - str);
1345                         }
1346                         if (toklen) {
1347                                 snprintf(buffer, sizeof(buffer), "'%.*s' %.*s", len, str, toklen, tok1);
1348                         } else {
1349                                 snprintf(buffer, sizeof(buffer), "'%.*s'", len, str);
1350                         }
1351                         yystpcpy(yyres, buffer);
1352                         return len + (toklen ? toklen + 1 : 0) + 2;
1353                 }               
1354         }       
1355         if (*yystr == '"') {
1356                 YYSIZE_T yyn = 0;
1357                 const char *yyp = yystr;
1358 
1359                 for (; *++yyp != '"'; ++yyn) {
1360                         yyres[yyn] = *yyp;
1361                 }
1362                 yyres[yyn] = '\0';
1363                 return yyn;
1364         }
1365         yystpcpy(yyres, yystr);
1366         return strlen(yystr);
1367 }
