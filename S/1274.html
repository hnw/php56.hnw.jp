  43 _zip_error_clear(struct zip_error *err)

  72 void
  73 _zip_error_get(const struct zip_error *err, int *zep, int *sep)
  74 {
  75     if (zep)
  76         *zep = err->zip_err;
  77     if (sep) {
  78         if (zip_error_get_sys_type(err->zip_err) != ZIP_ET_NONE)
  79             *sep = err->sys_err;
  80         else
  81             *sep = 0;
  82     }
  83 }
