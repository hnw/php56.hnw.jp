 169 typedef enum {

 173         STREAM_CRYPTO_METHOD_SSLv23_CLIENT = ((1 << 3) | (1 << 4) | (1 << 5) | 1),
 174         STREAM_CRYPTO_METHOD_TLSv1_0_CLIENT = (1 << 3 | 1),
